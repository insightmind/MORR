\begin{class}{Utility}

	\clsdsp{Contains the utility methods.}
	
	\begin{methods}
		\begin{method}{public static void SetAndDispatch(ref bool variable, bool value, Action onTrue, Action onFalse)}{Sets a boolean property and dispatches based on its value.}
			\begin{parameters}
				\para{ref bool variable}{The property to set}

				\para{Action onFalse}{The action to execute if variable was not set to false before calling this method, but value is false}
			\end{parameters}
		\end{method}
		\begin{method}{public static Type? GetTypeFromAnyAssembly(string type)}{Attempts to load the type with the specified name from any currently loaded assembly.}
			\begin{parameters}
				\para{string type}{The name of the type to load}
			\end{parameters}
			\return{Type?}{The Type with the corresponding name or null on failure}
		\end{method}
	\end{methods}
