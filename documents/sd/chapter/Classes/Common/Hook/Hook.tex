\subsection*{MORR.Shared.Hook}

\begin{class}{GlobalHook}
	\clsdiagram[width=0.6\textwidth]{resources/Classes/Common/Shared/Hook/GlobalHook.png}
	
	\clsdcl{public static class GlobalHook}
	
	\clsdsp{}
	
	\begin{attributes}
		\attribute{public static bool IsActive}{Represents if the GlobalHook is active or not.}
	\end{attributes}
	
	\begin{methods}					
			
		\begin{method}{public static void AddListener(RetrieveMessageCallBack callback, params MessageType[] types)}{Add a listener to multiple message types.}
			\begin{parameters}
				\para{RetrieveMessageCallBack callback}{The RetrieveMessageCallBack function to be called on}
				\para{params MessageType[] types}{An array of the types to listen for}	
			\end{parameters}
		\end{method}
		
		\begin{method}{public static void RemoveListener(RetrieveMessageCallBack callback, params MessageType[] types)}{Remove a listener from multiple message types.}
			\begin{parameters}
				\para{RetrieveMessageCallBack callback}{The RetrieveMessageCallBack to remove}
				\para{params MessageType[] types}{An array of the types to remove the callback for}	
			\end{parameters}
		\end{method}
		
		\begin{method}{public static void Initialize()}{Initializes the Library with the default hook.}
		\end{method}
		
		\begin{method}{public static void Initialize(IHookNativeMethods hook)}{Initializes the Library with the given hook.}
			\begin{parameters}
				\para{IHookNativeMethods hook}{The hook used as an interaction point.}
			\end{parameters}
		\end{method}
		
		\begin{method}{public static void FreeLibrary()}{Free the loaded library. To be called when this application terminates.}
		\end{method}
	\end{methods}
	
	\begin{enum}{public enum MessageType}
	
	\clsdsp{Subset of window messages.}
		
	\end{enum}
\end{class}
