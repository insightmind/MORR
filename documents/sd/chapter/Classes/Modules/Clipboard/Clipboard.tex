\begin{enum}{InteractionType}
    \clsdiagram[scale=0.9]{resources/Classes/Modules/Clipboard/Events/InteractionType.png}

    \clsdcl{public enum InteractionType}

    \clsdsp{An Enumeration to specify the four interaction type with a clipboard: clear, copy, cut and paste.}

    \begin{elements}
        \element{Clear}{Clear the content from the clipboard.}
        \element{Copy}{Copy content into the clipboard.}
        \element{Cut}{Cut content into the clipboard.}
        \element{Paste}{Paste from the clipboard.}
    \end{elements}
\end{enum}
